<objective>
Create a generic, pure-SQL KPI aggregation helper that can be called for any date window (or no window at all) and returns the triple (total_revenue, avg_order_value, total_orders) OR None when zero rows match. Purpose: Isolate pure SQL from endpoint wiring so we can unit-test the aggregation independently of FastAPI / query param parsing / response serialization.
Output: One new service module + one unit test file.
</objective>

    ```python
    from datetime import date
    from decimal import Decimal
    from sqlalchemy import func, select
    from sqlalchemy.ext.asyncio import AsyncSession
    from app.models import SalesRecord


    async def aggregate_kpi_summary(
        session: AsyncSession,
        start_date: date | None = None,
        end_date: date | None = None,
    ) -> dict | None:
        """Generic KPI aggregation over sales_records.

        Mirrors the WHERE total_value > 0 filter from the legacy summary endpoint
        (routers/kpis.py) so comparison windows are semantically identical.

        Returns None when no rows match — this distinguishes "no data" from
        "legitimate zero" per DELTA-05. Callers serialize None to the
        nullable `previous_period` / `previous_year` field (or null chart series).
        """
        stmt = (
            select(
                func.sum(SalesRecord.total_value).label("total_revenue"),
                func.avg(SalesRecord.total_value).label("avg_order_value"),
                func.count(SalesRecord.id).label("total_orders"),
            )
            .where(SalesRecord.total_value > 0)
        )
        if start_date is not None:
            stmt = stmt.where(SalesRecord.order_date >= start_date)
        if end_date is not None:
            stmt = stmt.where(SalesRecord.order_date <= end_date)

        row = (await session.execute(stmt)).one()
        if (row.total_orders or 0) == 0:
            return None
        return {
            "total_revenue": row.total_revenue or Decimal("0"),
            "avg_order_value": row.avg_order_value or Decimal("0"),
            "total_orders": int(row.total_orders),
        }
    ```

    Do NOT touch `backend/app/routers/kpis.py` in this task — integration happens in plan 08-02.
    Do NOT introduce any new Python date libraries (per CONTEXT cross-plan rules).

    Example seed pattern:
    ```python
    from datetime import date
    from decimal import Decimal
    import pytest
    from sqlalchemy import delete
    from app.database import AsyncSessionLocal
    from app.models import SalesRecord, UploadBatch
    from app.services.kpi_aggregation import aggregate_kpi_summary


    async def _seed(session, filename, rows):
        from datetime import datetime, timezone
        batch = UploadBatch(
            filename=filename,
            uploaded_at=datetime.now(timezone.utc),
            row_count=len(rows),
            error_count=0,
            status="success",
        )
        session.add(batch)
        await session.flush()
        for idx, (d, v) in enumerate(rows):
            session.add(SalesRecord(
                upload_batch_id=batch.id,
                order_number=f"{filename}-{idx}",
                order_date=d,
                total_value=v,
            ))
        await session.commit()
        return batch.id


    async def _cleanup(session, batch_id):
        await session.execute(delete(SalesRecord).where(SalesRecord.upload_batch_id == batch_id))
        await session.execute(delete(UploadBatch).where(UploadBatch.id == batch_id))
        await session.commit()


    @pytest.mark.asyncio
    async def test_bounded_window_happy_path():
        async with AsyncSessionLocal() as session:
            bid = await _seed(session, "agg-happy", [
                (date(2026, 3, 5), Decimal("100")),
                (date(2026, 3, 15), Decimal("200")),
                (date(2026, 3, 25), Decimal("300")),
                (date(2026, 4, 1), Decimal("999")),  # outside window
            ])
            try:
                result = await aggregate_kpi_summary(session, date(2026, 3, 1), date(2026, 3, 31))
                assert result is not None
                assert result["total_orders"] == 3
                assert result["total_revenue"] == Decimal("600")
            finally:
                await _cleanup(session, bid)
    ```
