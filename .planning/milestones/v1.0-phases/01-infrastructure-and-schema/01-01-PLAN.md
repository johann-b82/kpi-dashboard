**`backend/app/database.py`** -- async engine + session factory + Base:
```python
import os

from sqlalchemy.ext.asyncio import AsyncSession, create_async_engine
from sqlalchemy.orm import DeclarativeBase, sessionmaker

DATABASE_URL = (
    f"postgresql+asyncpg://{os.environ['POSTGRES_USER']}:"
    f"{os.environ['POSTGRES_PASSWORD']}@db:5432/{os.environ['POSTGRES_DB']}"
)

engine = create_async_engine(DATABASE_URL, echo=False)
AsyncSessionLocal = sessionmaker(engine, class_=AsyncSession, expire_on_commit=False)


class Base(DeclarativeBase):
    pass


async def get_async_db_session():
    async with AsyncSessionLocal() as session:
        yield session
```

IMPORTANT: Alembic's `%(POSTGRES_USER)s` syntax reads from environment variables automatically. This is Alembic/ConfigParser interpolation, NOT Python string formatting. It works because Alembic calls `config.get_section()` which resolves `%(ENV_VAR)s` from `os.environ` when `env.py` sets `config.set_section_option`. However, the simpler approach that works out of the box: override the URL in `env.py` using the async engine config. The `alembic.ini` URL serves as fallback. The `env.py` below handles this correctly by constructing the URL from env vars directly.

**`backend/alembic/env.py`** -- async migration runner, imports models for autogenerate:
```python
import asyncio
import os
from logging.config import fileConfig

from alembic import context
from sqlalchemy import pool
from sqlalchemy.engine import Connection
from sqlalchemy.ext.asyncio import async_engine_from_config

from app.models import Base  # noqa: F401 -- registers models with metadata

config = context.config

# Override sqlalchemy.url with env vars (sync URL for Alembic)
db_url = (
    f"postgresql+asyncpg://{os.environ['POSTGRES_USER']}:"
    f"{os.environ['POSTGRES_PASSWORD']}@db:5432/{os.environ['POSTGRES_DB']}"
)
config.set_main_option("sqlalchemy.url", db_url)

if config.config_file_name is not None:
    fileConfig(config.config_file_name)

target_metadata = Base.metadata


def do_run_migrations(connection: Connection) -> None:
    context.configure(connection=connection, target_metadata=target_metadata)
    with context.begin_transaction():
        context.run_migrations()


async def run_async_migrations() -> None:
    connectable = async_engine_from_config(
        config.get_section(config.config_ini_section, {}),
        prefix="sqlalchemy.",
        poolclass=pool.NullPool,
    )
    async with connectable.connect() as connection:
        await connection.run_sync(do_run_migrations)
    await connectable.dispose()


def run_migrations_online() -> None:
    asyncio.run(run_async_migrations())


if context.is_offline_mode():
    raise NotImplementedError("Offline migrations not supported")
else:
    run_migrations_online()
```

Note: The `env.py` overrides `sqlalchemy.url` with an async URL (`postgresql+asyncpg://`) because it uses `async_engine_from_config`. This is the correct pattern from the official Alembic async template -- the async engine handles the connection, then `run_sync` executes the sync migration DDL.
