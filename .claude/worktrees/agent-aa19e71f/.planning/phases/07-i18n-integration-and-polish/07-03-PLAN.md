    Create `frontend/src/contexts/SettingsDraftContext.tsx` exposing the lightest-possible dirty-state contract. Verbatim structure:

    ```tsx
    import { createContext, useContext, useState, type ReactNode } from "react";

    interface SettingsDraftStatus {
      isDirty: boolean;
      setDirty: (v: boolean) => void;
    }

    const Ctx = createContext<SettingsDraftStatus | null>(null);

    export function SettingsDraftProvider({ children }: { children: ReactNode }) {
      const [isDirty, setDirty] = useState(false);
      return <Ctx.Provider value={{ isDirty, setDirty }}>{children}</Ctx.Provider>;
    }

    // Returns null when the consumer is outside the provider (impossible if App wraps).
    // Consumers read .isDirty to decide disable state; null-coalescing defaults to enabled.
    export function useSettingsDraftStatus(): SettingsDraftStatus | null {
      return useContext(Ctx);
    }
    ```

    In `frontend/src/App.tsx`, import `SettingsDraftProvider` and wrap the entire component tree that contains BOTH `<NavBar />` and the route outlet (including `<SettingsPage />`). The provider must be an ancestor of both — otherwise NavBar cannot read the status. Do not wrap just NavBar, do not wrap just SettingsPage.

    Do not use Zustand, event emitters, or window-level globals. No new dependency.
