    **Edit `backend/app/routers/kpis.py`**: Extract a private helper `_bucketed_series(db, start, end, granularity) -> list[tuple[date, Decimal]]` that runs the existing bucket SQL and returns a list of (bucket_date, revenue) tuples in order. Then restructure `get_chart_data`:

    ```python
    import asyncio
    from app.schemas import ChartPoint, ChartResponse, KpiSummary, KpiSummaryComparison, LatestUploadResponse

    _TRUNC_MAP = {"daily": "day", "weekly": "week", "monthly": "month"}


    async def _bucketed_series(
        db: AsyncSession,
        start: date | None,
        end: date | None,
        granularity: Literal["daily", "weekly", "monthly"],
    ) -> list[tuple[date, Decimal]]:
        bucket = func.date_trunc(_TRUNC_MAP[granularity], SalesRecord.order_date).label("bucket")
        stmt = (
            select(bucket, func.sum(SalesRecord.total_value).label("revenue"))
            .where(SalesRecord.total_value > 0)
            .where(SalesRecord.order_date.isnot(None))
            .group_by(bucket)
            .order_by(bucket)
        )
        if start is not None:
            stmt = stmt.where(SalesRecord.order_date >= start)
        if end is not None:
            stmt = stmt.where(SalesRecord.order_date <= end)
        result = await db.execute(stmt)
        return [(row.bucket.date(), row.revenue or Decimal("0")) for row in result.all()]


    @router.get("/chart", response_model=ChartResponse)
    async def get_chart_data(
        start_date: date | None = Query(None),
        end_date: date | None = Query(None),
        granularity: Literal["daily", "weekly", "monthly"] = Query("monthly"),
        prev_start: date | None = Query(None),
        prev_end: date | None = Query(None),
        comparison: Literal["previous_period", "previous_year", "none"] = Query("none"),
        db: AsyncSession = Depends(get_async_db_session),
    ) -> ChartResponse:
        want_prior = (
            comparison != "none"
            and prev_start is not None
            and prev_end is not None
        )

        if want_prior:
            current_rows, prior_rows = await asyncio.gather(
                _bucketed_series(db, start_date, end_date, granularity),
                _bucketed_series(db, prev_start, prev_end, granularity),
            )
        else:
            current_rows = await _bucketed_series(db, start_date, end_date, granularity)
            prior_rows = None

        current = [ChartPoint(date=d.isoformat(), revenue=r) for d, r in current_rows]

        previous: list[ChartPoint] | None = None
        if prior_rows is not None:
            # Positional alignment: prior[i] → current_rows[i] (X-axis rewrite for overlay).
            # CHART-03: when prior has fewer buckets than current, emit revenue=None trailing buckets.
            # Trailing prior buckets beyond current's range are dropped (current is authoritative).
            aligned: list[ChartPoint] = []
            for i, (current_date, _) in enumerate(current_rows):
                if i < len(prior_rows):
                    aligned.append(ChartPoint(date=current_date.isoformat(), revenue=prior_rows[i][1]))
                else:
                    aligned.append(ChartPoint(date=current_date.isoformat(), revenue=None))
            previous = aligned

        return ChartResponse(current=current, previous=previous)
    ```

    IMPORTANT notes for the executor:
    - Do NOT introduce `dateutil.relativedelta` or any new date library (CONTEXT cross-plan rule).
    - The positional-alignment strategy is simpler than computing date arithmetic shifts and still satisfies CHART-01 (aligned X axis) + CHART-03 (null gaps for partial prior data). Recharts renders against the shared `current_rows` date domain.
    - If `current_rows` is empty (current window has zero data), `previous` becomes an empty list (not None) when `want_prior` is true. That's acceptable — the frontend can render nothing.
    - Preserve the existing `Literal` import from `typing`.
